UNION ALL
SELECT DEPTNO FROM DEPARTMENT

11d)SELECT * FROM EMPLOYEE

11e)SELECT EMPNAME,SALARY FROM EMPLOYEE
WHERE SALARY > (SELECT MIN(SALARY) FROM EMPLOYEE) AND JOB LIKE 'A%');
